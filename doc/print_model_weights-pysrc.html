14  if __name__ == '__main__': 
15      parser = argparse.ArgumentParser(description="Prints out the weights of a \

18                                       formatter_class=argparse.ArgumentDefaultsHelpFormatter) 
19      parser.add_argument('model_file', help='model file to load') 
20      parser.add_argument('--k', help='number of top features to print (0 for all)',

22      args = parser.parse_args() 
23   
24      k = args.k if args.k > 0 else None 
25   
26      learner = Learner() 
27      learner.load_model(args.model_file) 
28      weights = learner.get_model_params() 
29   
30      print("number of nonzero features:", len(weights), file=sys.stderr) 
31   
32      for feat, val in sorted(weights.items(), key=lambda x: -abs(x[1]))[:k]: 
33          print("{:.12f}\t{}".format(val, feat))
